Creator Videos Bookmarklet

Drag the link below to your bookmarks bar. When on a creator page with the Videos panel, click it to console.log each active video as { name: { Video ID, Link, Posted, Duration, Affiliate GMV, Items sold, Est. commissions, Direct GMV, RPM, Views, Completion rate, Details } }. Inactive videos are skipped.

javascript:(function(){
  var items = document.querySelectorAll('.arco-space-item');
  var root = items[items.length - 1];
  if (!root) { console.warn('Videos container not found'); return; }

  var rows = root.querySelectorAll('tbody tr.arco-table-tr');
  rows.forEach(function(row){
    // Active videos have a TikTok video link with these classes; inactive rows do not.
    var link = row.querySelector('a.text-12.hover\\:underline');
    if (!link) return;

    var linkParent = link.parentElement;          // .flex.items-center wrapping the link
    var infoCol   = linkParent.parentElement;     // .ml-6.flex-1.overflow-auto column

    // Video name lives in parent-of-parent of the link.
    var nameContainer = infoCol.querySelector('.whitespace-nowrap.overflow-ellipsis.overflow-hidden');
    var nameEl = nameContainer && nameContainer.querySelector('.overflow-hidden.overflow-ellipsis');
    var name = nameEl ? nameEl.textContent.replace(/\s+/g, ' ') : '';

    // Video ID lives in the sibling div immediately above the link.
    var idText = '';
    var idDiv = linkParent.previousElementSibling;
    if (idDiv) {
      var idEl = idDiv.querySelector('.text-12');
      idText = idEl ? idEl.textContent.trim().replace(/^Video ID:\s*/, '') : '';
    }

    // Posted & Duration live in the sibling div immediately below the link.
    var posted = '', duration = '';
    var postedDiv = linkParent.nextElementSibling;
    if (postedDiv) {
      var pEl = postedDiv.querySelector('.text-12');
      if (pEl) {
        var m = pEl.textContent.trim().match(/Posted:\s*(\S+)\s+Duration:\s*(\S+)/);
        if (m) { posted = m[1]; duration = m[2]; }
      }
    }

    // Per-row table cells: 0=Info, 1=Affiliate GMV, 2=Items sold, 3=Est. commissions,
    // 4=Direct GMV, 5=RPM, 6=Views, 7=Completion rate, 8=Video details (View link).
    var tds = row.querySelectorAll('td.arco-table-td');
    var get = function(td){
      var inner = td.querySelector('.arco-table-cell-wrap-value');
      return inner ? inner.textContent.trim() : '';
    };

    var detailsLink = '';
    if (tds[8]) {
      var a = tds[8].querySelector('a');
      detailsLink = a ? a.getAttribute('href') : '';
    }

    var o = {};
    o[name] = {
      'Video ID':         idText,
      'Link':             link.getAttribute('href'),
      'Posted':           posted,
      'Duration':         duration,
      'Affiliate GMV':    tds[1] ? get(tds[1]) : '',
      'Items sold':       tds[2] ? get(tds[2]) : '',
      'Est. commissions': tds[3] ? get(tds[3]) : '',
      'Direct GMV':       tds[4] ? get(tds[4]) : '',
      'RPM':              tds[5] ? get(tds[5]) : '',
      'Views':            tds[6] ? get(tds[6]) : '',
      'Completion rate':  tds[7] ? get(tds[7]) : '',
      'Details':          detailsLink
    };
    console.log(o);
  });
})();
