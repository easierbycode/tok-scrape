Key Metrics & Videos Bookmarklet

Drag the link below to your bookmarks bar. When on a page with the creator selector, Key metrics panel, and Videos table, click it to console.log each Key metric (tagged with the selected creator and vs. previous month trend) followed by each active video in the Videos table.

javascript:(function(){
  var UP = "M2.97731 0.130983C3.11029 -0.0430998 3.37217 -0.0437687 3.50604 0.129632L6.44523 3.93668C6.6144 4.15581 6.45821 4.47372 6.18138 4.47372H4.60189V8.33326C4.60189 8.51735 4.45265 8.66659 4.26855 8.66659H2.26855C2.08446 8.66659 1.93522 8.51735 1.93522 8.33326V4.47372H0.333945C0.057903 4.47372 -0.0985142 4.15739 0.0690572 3.93803L2.97731 0.130983Z";
  var items = document.querySelectorAll('.arco-space-item');
  var metricsRoot = items[2];
  var checked = items[1] && items[1].querySelector('.arco-radio-checked');
  var card = checked && checked.closest('.flex.w-full.flex-col');
  var nameEl = card && card.querySelector('.text-body-s-medium');
  var creator = nameEl ? nameEl.textContent.trim() : '';

  if (metricsRoot) {
    metricsRoot.querySelectorAll('.flex.flex-col.items-start').forEach(function(c){
      var k = c.querySelector('.text-14'),
          v = c.querySelector('.text-20');
      if (!k || !v) return;
      var row = v.nextElementSibling,
          p = row && row.querySelector('svg path'),
          pct = row && row.querySelector('span');
      var dir = p && p.getAttribute('d') === UP ? '\u2191' : '\u2193';
      var o = { creator: creator };
      o[k.textContent.trim()] = v.textContent.trim();
      o['vs. previous month'] = dir + ' ' + (pct ? pct.textContent.trim() : '');
      console.log(o);
    });
  } else { console.warn('Key metrics container not found'); }

  // Active videos in the last .arco-space-item (Videos table)
  var videosRoot = items[items.length - 1];
  if (!videosRoot) { console.warn('Videos container not found'); return; }
  videosRoot.querySelectorAll('tr.arco-table-tr').forEach(function(row){
    var link = row.querySelector('a.text-12.hover\\:underline');
    if (!link) return; // inactive videos have no link — skip
    var lp = link.parentElement,
        lg = lp.parentElement;
    var nm = lg.querySelector('.overflow-hidden.overflow-ellipsis');
    var idD = lp.previousElementSibling,
        idE = idD && idD.querySelector('.text-12');
    var pdD = lp.nextElementSibling,
        pdE = pdD && pdD.querySelector('.text-12');
    var vid = (idE ? idE.textContent : '').replace(/^\s*Video ID:\s*/i, '').trim();
    var pdT = (pdE ? pdE.textContent : '').replace(/\s+/g, ' ').trim();
    var m = pdT.match(/Posted:\s*(\S+)\s+Duration:\s*(\S+)/);
    var cells = row.querySelectorAll('td.arco-table-td');
    var dA = cells[8] && cells[8].querySelector('a');
    var nameText = nm ? nm.textContent.replace(/\s+/g, ' ') : '';
    var o = {};
    o[nameText] = {
      'Video ID': vid,
      'Link': link.href,
      'Posted': m ? m[1] : '',
      'Duration': m ? m[2] : '',
      'Affiliate GMV': cells[1] ? cells[1].textContent.trim() : '',
      'Items sold': cells[2] ? cells[2].textContent.trim() : '',
      'Est. commissions': cells[3] ? cells[3].textContent.trim() : '',
      'Direct GMV': cells[4] ? cells[4].textContent.trim() : '',
      'RPM': cells[5] ? cells[5].textContent.trim() : '',
      'Views': cells[6] ? cells[6].textContent.trim() : '',
      'Completion rate': cells[7] ? cells[7].textContent.trim() : '',
      'details link': dA ? dA.href : ''
    };
    console.log(o);
  });
})();
