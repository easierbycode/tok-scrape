Key Metrics Bookmarklet

Drag the link below to your bookmarks bar. When on a page with the creator selector, Key metrics panel, and Videos table, click it to console.log each metric (tagged with the selected creator and vs. previous month trend) and each active creator video with its stats.

Log Key Metrics

javascript:(function(){
  var UP = "M2.97731 0.130983C3.11029 -0.0430998 3.37217 -0.0437687 3.50604 0.129632L6.44523 3.93668C6.6144 4.15581 6.45821 4.47372 6.18138 4.47372H4.60189V8.33326C4.60189 8.51735 4.45265 8.66659 4.26855 8.66659H2.26855C2.08446 8.66659 1.93522 8.51735 1.93522 8.33326V4.47372H0.333945C0.057903 4.47372 -0.0985142 4.15739 0.0690572 3.93803L2.97731 0.130983Z";
  var items = document.querySelectorAll('.arco-space-item');
  var metricsRoot = items[2];
  if (!metricsRoot) { console.warn('Key metrics container not found'); return; }

  var checked = items[1] && items[1].querySelector('.arco-radio-checked');
  var card = checked && checked.closest('.flex.w-full.flex-col');
  var nameEl = card && card.querySelector('.text-body-s-medium');
  var creator = nameEl ? nameEl.textContent.trim() : '';

  metricsRoot.querySelectorAll('.flex.flex-col.items-start').forEach(function(c){
    var k = c.querySelector('.text-14'),
        v = c.querySelector('.text-20');
    if (!k || !v) return;
    var row = v.nextElementSibling,
        p = row && row.querySelector('svg path'),
        pct = row && row.querySelector('span');
    var dir = p && p.getAttribute('d') === UP ? '\u2191' : '\u2193';
    var o = { creator: creator };
    o[k.textContent.trim()] = v.textContent.trim();
    o['vs. previous month'] = dir + ' ' + (pct ? pct.textContent.trim() : '');
    console.log(o);
  });

  // Active creator videos (under the last .arco-space-item)
  var videosRoot = items[items.length - 1];
  if (!videosRoot || videosRoot === metricsRoot) return;
  videosRoot.querySelectorAll('a.text-12.hover\\:underline').forEach(function(a){
    var linkRow = a.parentElement;
    var idRow = linkRow.previousElementSibling;
    var pdRow = linkRow.nextElementSibling;
    var infoBlock = linkRow.parentElement;
    if (!idRow || !pdRow || !infoBlock) return;
    var idEl = idRow.querySelector('.text-12');
    var pdEl = pdRow.querySelector('.text-12');
    if (!idEl || !pdEl) return;
    var vid = idEl.textContent.replace(/\s+/g,' ').replace(/^\s*Video ID:\s*/,'').trim();
    var pdText = pdEl.textContent.replace(/\s+/g,' ').trim();
    var pm = pdText.match(/Posted:\s*(\S+)/);
    var dm = pdText.match(/Duration:\s*(\S+)/);
    var nm = infoBlock.querySelector('.overflow-hidden.overflow-ellipsis');
    var name = nm ? nm.textContent.replace(/\s+/g,' ').trim() : '';
    var tr = a.closest('tr');
    var tds = tr ? tr.querySelectorAll('td') : [];
    var cell = function(i){ return tds[i] ? tds[i].textContent.replace(/\s+/g,' ').trim() : ''; };
    var detailsA = tds.length ? tds[tds.length - 1].querySelector('a') : null;
    var o = {};
    o[name] = {
      'Video ID': vid,
      'Link': a.textContent.replace(/\s+/g,' ').trim(),
      'Posted': pm ? pm[1] : '',
      'Duration': dm ? dm[1] : '',
      'Affiliate GMV': cell(1),
      'Items sold': cell(2),
      'Est. commissions': cell(3),
      'Direct GMV': cell(4),
      'RPM': cell(5),
      'Views': cell(6),
      'Completion rate': cell(7),
      'Details': detailsA ? detailsA.href : ''
    };
    console.log(o);
  });
})();
