Key Metrics Bookmarklet

Drag the link below to your bookmarks bar. When on a page with the Key metrics panel, click it to console.log each metric.

Log Key Metrics

Source

javascript:(function(){
  var s = document.querySelectorAll('.arco-space-item')[2];
  if (!s) { console.warn('Key metrics container not found'); return; }
  s.querySelectorAll('.flex.flex-col.items-start').forEach(function(c){
    var k = c.querySelector('.text-14'),
        v = c.querySelector('.text-20');
    if (k && v) {
      var o = {};
      o[k.textContent.trim()] = v.textContent.trim();
      console.log(o);
    }
  });
})();
